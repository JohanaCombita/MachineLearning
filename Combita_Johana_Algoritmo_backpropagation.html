#Sigmoide: Función de activación usada en la regresión logística (0 y 1)
def sigmoid(x):
    return 1 / (1 + np.exp(-x)) 
#Derivada de la sigmoide: Se usa en el cálculo del gradiente para actualizar los pesos.
def d_sigmoid(x):
    return sigmoid(x)*(1-sigmoid(x))

# Implementación de Backpropagation con descenso de gradiente y una capa oculta
def backprop(a, b, h=5, epochs=500, eta=0.1):
    ### Inicializar matrices y listas
    np.random.seed(1302)
    neurons = [np.shape(a)[0], h, np.shape(b)[0]]  # [2,5,1]
    weights = [np.random.uniform(low=-1, high=1, size=(neurons[i], neurons[i+1])) for i in range(len(neurons)-1)]  # [(2,5), (5,1)]
    bias = [np.random.uniform(low=-1, high=1, size=(neurons[i+1], 1)) for i in range(len(neurons)-1)]  # [(5,1), (1,1)]
    errors = []
          
    for i in range(epochs):
        ### Propagar hacia delante
        # entrada ponderada de la capa oculta
        B_hidden = np.dot(weights[0].T, a) + bias[0]  # (5, N)
        hidden = sigmoid(B_hidden)  # (5, N)
        
        # entrada ponderada de la capa de salida
        B_out = np.dot(weights[1].T, hidden) + bias[1]  # (1, N)
        b_out = sigmoid(B_out)  # (1, N)
        
        # Calculo de gradiente
        # ERROR (Diferencia entre real y predicha)
        d_error = b_out - b 
        error = np.mean((d_error) ** 2)
        errors.append(error)
                
        # BACKPROPAGATION - capa de salida (pesos v y sesgos t)
        aux_delta = d_error * d_sigmoid(B_out)  # (1, N)
        v_delta_out = np.dot(hidden, aux_delta.T)  # (5,1)
        t_delta_out = aux_delta.sum(axis=1, keepdims=True)  # (1,1)
        
        # capa ocula (pesos w y sesgos b)
        aux_delta_w = np.dot(weights[1], aux_delta) * d_sigmoid(B_hidden)
        w_delta_hidden = np.dot(a, aux_delta_w.T)
        b_delta_hidden = aux_delta_w.sum(axis=1, keepdims=True)
        
        # Actualizar pesos y sesgos
        weights[0] -= eta * w_delta_hidden
        bias[0] -= eta * b_delta_hidden
        weights[1] -= eta * v_delta_out
        bias[1] -= eta * t_delta_out
        
        # Imprimir error cada 100 iteraciones
        if i % 100 == 0:
            print(f"Epoch {i}: Error = {error}")
            
    # Devolver valores aprendidos
    return weights, bias, errors

def backprop_with_momentum(a, b, h=5, epochs=1000, eta=0.5, alpha=0.9):
    # Inicializar matrices y listas
    np.random.seed(1302)
    neurons = [np.shape(a)[0], h, np.shape(b)[0]]  # [2,5,1]
    weights = [np.random.uniform(low=-1, high=1, size=(neurons[i], neurons[i+1])) for i in range(len(neurons)-1)]  # [(2,5), (5,1)]
    bias = [np.random.uniform(low=-1, high=1, size=(neurons[i+1], 1)) for i in range(len(neurons)-1)]  # [(5,1), (1,1)]
    errors = []
    
    # Inicializar cambios anteriores (inercia) en cero
    weight_updates = [np.zeros_like(w) for w in weights]
    bias_updates = [np.zeros_like(b) for b in bias]
    
    for i in range(epochs):
        # Propagación hacia adelante
        B_hidden = np.dot(weights[0].T, a) + bias[0]  # (5, N)
        hidden = sigmoid(B_hidden)  # (5, N)
        B_out = np.dot(weights[1].T, hidden) + bias[1]  # (1, N)
        b_out = sigmoid(B_out)  # (1, N)
        
        # Cálculo del error
        d_error = b_out - b 
        error = np.mean((d_error) ** 2)
        errors.append(error)
        
        # Retropropagación - capa de salida
        aux_delta = d_error * d_sigmoid(B_out)  # (1, N)
        v_delta_out = np.dot(hidden, aux_delta.T)  # (5,1)
        t_delta_out = aux_delta.sum(axis=1, keepdims=True)  # (1,1)
        
        # Retropropagación - capa oculta
        aux_delta_w = np.dot(weights[1], aux_delta) * d_sigmoid(B_hidden)
        w_delta_hidden = np.dot(a, aux_delta_w.T)
        b_delta_hidden = aux_delta_w.sum(axis=1, keepdims=True)
        
        # Actualización de pesos y sesgos con inercia
        weight_updates[0] = alpha * weight_updates[0] - eta * w_delta_hidden
        bias_updates[0] = alpha * bias_updates[0] - eta * b_delta_hidden
        weight_updates[1] = alpha * weight_updates[1] - eta * v_delta_out
        bias_updates[1] = alpha * bias_updates[1] - eta * t_delta_out
        
        weights[0] += weight_updates[0]
        bias[0] += bias_updates[0]
        weights[1] += weight_updates[1]
        bias[1] += bias_updates[1]
        
        # Imprimir error cada 100 iteraciones
        if i % 100 == 0:
            print(f"Epoch {i}: Error = {error}")
            
    # Devolver valores aprendidos
    return weights, bias, errors

Se establecen los mismos parámetros del ejercicio anterior y un alpha de 0.9, siendo el factor de inercia que controla cuanto de la actualización de los pesos se mantiene en una nueva actualización, suavizando los cambios de los pesos, evitando oscilaciones y acelerando la convergencia.

def backprop_gen_with_momentum(a, b, h, epochs=500, eta=0.1, alpha=0.9):
    # Inicialización de matrices
    np.random.seed(1302)
    neurons = [np.shape(a)[0]] + h + [np.shape(b)[0]] 
    weights = [np.random.uniform(low=-1, high=1, size=(neurons[i], neurons[i+1])) for i in range(len(neurons)-1)] 
    bias = [np.random.uniform(low=-1, high=1, size=(neurons[i+1], 1)) for i in range(len(neurons)-1)]
    
    # Inicialización de las actualizaciones anteriores (momentum)
    prev_weight_updates = [np.zeros_like(w) for w in weights]
    prev_bias_updates = [np.zeros_like(b) for b in bias]
    
    errors = []
    
    for i in range(epochs):
        # Propagación hacia adelante
        B_hidden = [np.dot(weights[0].T, a) + bias[0]]
        
        for j in range(1, len(h)):
            B_hidden.append(np.dot(weights[j].T, sigmoid(B_hidden[j-1])) + bias[j])
            
        hidden = [sigmoid(B_hidden[j]) for j in range(len(h))]
        
        # Entrada ponderada de la capa de salida
        B_out = np.dot(weights[-1].T, sigmoid(B_hidden[-1])) + bias[-1]  
        b_out = sigmoid(B_out)
        
        # Cálculo de gradiente
        # ERROR (Diferencia entre real y predicha)
        d_error = b_out - b
        error = np.mean((d_error) ** 2)
        errors.append(error)
        
        # BACKPROPAGATION
        # Capa de salida
        aux_delta = d_error * d_sigmoid(B_out)
        v_delta_out = np.dot(sigmoid(B_hidden[-1]), aux_delta.T)
        t_delta_out = aux_delta.sum(axis=1, keepdims=True)
        
        # Capas ocultas
        aux_delta_w = [np.dot(weights[-1], aux_delta) * d_sigmoid(B_hidden[-1])]
        w_delta_hidden = [np.dot(sigmoid(B_hidden[-2]) if len(h) > 1 else a, aux_delta_w[0].T)]
        b_delta_hidden = [aux_delta_w[0].sum(axis=1, keepdims=True)]
        
        # Resto de capas
        for j in range(1, len(h)):
            prev_activation = sigmoid(B_hidden[-j-2]) if j < len(h)-1 else a
            aux_delta_w.append(np.dot(weights[-1-j], aux_delta_w[-1]) * d_sigmoid(B_hidden[-1-j]))
            w_delta_hidden.append(np.dot(prev_activation, aux_delta_w[-1].T))
            b_delta_hidden.append(aux_delta_w[-1].sum(axis=1, keepdims=True))
        
        # Actualización de pesos y sesgos con momentum
        w_delta_hidden = w_delta_hidden[::-1] 
        b_delta_hidden = b_delta_hidden[::-1]
        
        for j in range(len(h)):
            prev_weight_updates[j] = alpha * prev_weight_updates[j] - eta * w_delta_hidden[j]
            prev_bias_updates[j] = alpha * prev_bias_updates[j] - eta * b_delta_hidden[j]
            
            weights[j] += prev_weight_updates[j]
            bias[j] += prev_bias_updates[j]
        
        prev_weight_updates[-1] = alpha * prev_weight_updates[-1] - eta * v_delta_out
        prev_bias_updates[-1] = alpha * prev_bias_updates[-1] - eta * t_delta_out
        
        weights[-1] += prev_weight_updates[-1]
        bias[-1] += prev_bias_updates[-1]
        
        # Imprimir error cada 100 iteraciones
        if i % 100 == 0:
            print(f"Epoch {i}: Error = {error}")
            
    # Devolver los valores aprendidos
    return weights, bias, errors

def visualizar_frontera(pesos_finales, sesgos_finales, lin_spi, ceros, unos):
    # Crear grid de puntos
    X1, X2 = np.meshgrid(np.linspace(-1, 1, 100), np.linspace(-1, 1, 100))
    entrada = np.vstack([X1.ravel(), X2.ravel()])
    
    # Forward propagation a través de todas las capas
    Z = entrada
    for w, b in zip(pesos_finales, sesgos_finales):
        Z = sigmoid(np.dot(w.T, Z) + b)
    
    # Visualización
    plt.figure(figsize=[4, 4])
    p = plt.pcolormesh(X1, X2, Z.reshape(X1.shape), cmap='seismic', vmin=0, vmax=1)
    plt.scatter(lin_spi['x1'][ceros], lin_spi['x2'][ceros], color='r', s=20, label='Clase 0')
    plt.scatter(lin_spi['x1'][unos], lin_spi['x2'][unos], color='g', s=20, label='Clase 1')
    plt.xlim([-1, 1])
    plt.ylim([-1, 1])
    plt.colorbar(p)
    plt.legend()
    plt.title('Frontera de Decisión')
    plt.xlabel('x1')
    plt.ylabel('x2')
    plt.show()
